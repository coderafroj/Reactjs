// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract FreelancePlatform {
    struct Job {
        address employer;
        address freelancer;
        uint256 amount;
        bool completed;
        bool paid;
    }

    mapping(uint256 => Job) public jobs;
    uint256 public jobCount;

    event JobCreated(uint256 indexed jobId, address employer, address freelancer, uint256 amount);
    event JobCompleted(uint256 indexed jobId, address freelancer);
    event PaymentSent(uint256 indexed jobId, address freelancer, uint256 amount);

    function createJob(address _freelancer, uint256 _amount) external {
        jobCount++;
        jobs[jobCount] = Job({
            employer: msg.sender,
            freelancer: _freelancer,
            amount: _amount,
            completed: false,
            paid: false
        });
        emit JobCreated(jobCount, msg.sender, _freelancer, _amount);
    }

    function completeJob(uint256 _jobId) external {
        require(jobs[_jobId].freelancer == msg.sender, "Only freelancer can complete job");
        jobs[_jobId].completed = true;
        // Send payment
        payable(jobs[_jobId].freelancer).transfer(jobs[_jobId].amount);
        jobs[_jobId].paid = true;
        emit JobCompleted(_jobId, msg.sender);
        emit PaymentSent(_jobId, jobs[_jobId].freelancer, jobs[_jobId].amount);
    }
}

    uint256 public tokenCounter;

    constructor() ERC721("NFT", "NFT") {
        tokenCounter = 0;
    }

    function mint(string memory _tokenUri) public {
        tokenCounter++;
        _safeMint(msg.sender, tokenCounter);
        tokenToUri[tokenCounter] = _tokenUri;

        require(_tokenId <= tokenCounter, "Token ID does not exist");

import React, { useState } from "react";
import { ethers } from "ethers";

function App() {
  const [nftName, setNftName] = useState("");
  const [nftImage, setNftImage] = useState("");

  const mintNFT = async () => {
    const provider = new ethers.providers.Web3Provider(window.ethereum);
    const signer = provider.getSigner();
    const contract = new ethers.Contract("YOUR_CONTRACT_ADDRESS", ABI, signer);

    const transaction = await contract.mint(nftImage);
    await transaction.wait();
    alert("NFT minted!");
  };

  return (
    <div>
      <input
        type="text"
        value={nftName}
        onChange={(e) => setNftName(e.target.value)}
        placeholder="NFT Name"
      />
      <input
        type="text"
        value={nftImage}
        onChange={(e) => setNftImage(e.target.value)}
        placeholder="NFT Image URL"
      />
      <button onClick={mintNFT}>Mint NFT</button>
    </div>
  );
}

export default App;
